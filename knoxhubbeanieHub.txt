local camNoClipEnabled=false

local camPitch=math.rad(-20)
local camYaw=0

local function enableCamNoClip()
	if camNoClipEnabled then return end
	camNoClipEnabled=true
	local cam=workspace.CurrentCamera

	local char=lp.Character
	local hrp=char and char:FindFirstChild("HumanoidRootPart")
	if hrp then
		local look=cam.CFrame.Position-hrp.Position
		camDist=math.clamp(look.Magnitude,3,50)
		camYaw=math.atan2(-look.X,-look.Z)
		camPitch=math.atan2(look.Y,math.sqrt(look.X^2+look.Z^2))
	end
	cam.CameraType=Enum.CameraType.Scriptable
	camNoClipConn=RS.RenderStepped:Connect(function()
		if not camNoClipEnabled then return end
		local c2=lp.Character
		local h2=c2 and c2:FindFirstChild("HumanoidRootPart")
		if not h2 then return end
		local delta=UIS:GetMouseDelta()
		camYaw=camYaw-delta.X*0.004
		camPitch=math.clamp(camPitch-delta.Y*0.004,math.rad(-80),math.rad(80))
		local offset=V3(
			math.sin(camYaw)*math.cos(camPitch),
			math.sin(camPitch),
			math.cos(camYaw)*math.cos(camPitch)
		)*camDist
		local focus=h2.Position+V3(0,1.5,0)
		cam.CFrame=CFrame.lookAt(focus+offset,focus)
	end)

	camScrollConn=UIS.InputChanged:Connect(function(inp)
		if not camNoClipEnabled then return end
		if inp.UserInputType==Enum.UserInputType.MouseWheel then
			camDist=math.clamp(camDist-inp.Position.Z*1.5,3,80)
		end
	end)
end

local function disableCamNoClip()
	if not camNoClipEnabled then return end
	camNoClipEnabled=false
	if camNoClipConn then camNoClipConn:Disconnect();camNoClipConn=nil end
	if camScrollConn then camScrollConn:Disconnect();camScrollConn=nil end
	workspace.CurrentCamera.CameraType=Enum.CameraType.Custom
end

local function isValidStealPrompt(prompt)
	if not prompt or not prompt.Parent or not prompt.Enabled then return false end
	local state=prompt:GetAttribute("State")
	local actionText=prompt.ActionText
	return state=="Steal" or state=="Grab" or actionText=="Steal" or actionText=="Grab"
end

local function findNearestSteal()
	local char=lp.Character
	local root=char and (char:FindFirstChild("HumanoidRootPart") or char:FindFirstChild("UpperTorso"))
	if not root then return nil end
	local plots=workspace:FindFirstChild("Plots");if not plots then return nil end
	local nearest,minDist=nil,150
	for _,plot in ipairs(plots:GetChildren())do
		if isMyPlot(plot) then continue end
		local podiums=plot:FindFirstChild("AnimalPodiums");if not podiums then continue end
		for _,podium in ipairs(podiums:GetChildren())do
			local base=podium:FindFirstChild("Base")
			local spawnPoint=base and base:FindFirstChild("Spawn")
			local attachment=spawnPoint and spawnPoint:FindFirstChild("PromptAttachment")
			if attachment then
				for _,child in ipairs(attachment:GetChildren())do
					if child:IsA("ProximityPrompt") and isValidStealPrompt(child) then
						local d=(root.Position-child.Parent.WorldPosition).Magnitude
						if d<minDist then minDist=d;nearest=child end
					end
				end
			end
		end
	end
	return nearest
end

local function firePromptConnections(prompt,signalName)
	local ok,conns=pcall(getconnections,prompt[signalName])
	if not ok or not conns then return end
	for _,conn in ipairs(conns)do
		if conn.Function then task.spawn(conn.Function) end
	end
end

local function parseBrainrotValue(text)
	text=tostring(text or ""):gsub("%s","")
	local num,suffix=text:match("([%d%.]+)([KkMmBbTt]?)")
	if not num then return 0 end
	num=tonumber(num)or 0
	local mult={K=1e3,M=1e6,B=1e9,T=1e12}
	return num*(mult[(suffix or ""):upper()]or 1)
end
local function destroyBrainrotBeam()
	if brainrotBeam then brainrotBeam:Destroy();brainrotBeam=nil end
	if brainrotBAtt0 then brainrotBAtt0:Destroy();brainrotBAtt0=nil end

local function scanBrainrotDebris()
	if not brainrotEnabled then return end
	local debris=workspace:FindFirstChild("Debris");if not debris then return end
	local best={value=-1,part=nil,displayText="",valueText=""}
	for _,template in ipairs(debris:GetChildren())do
		if template.Name=="FastOverheadTemplate"then
			local surfaceGui=template:FindFirstChildOfClass("SurfaceGui");if not surfaceGui then continue end
			local genLabel=surfaceGui:FindFirstChild("Generation",true)
			if not genLabel or not genLabel:IsA("TextLabel")then continue end
			local text=genLabel.Text or ""
			if text~=""and(text:find("/s")or text:find("K")or text:find("M")or text:find("B"))then
				local val=parseBrainrotValue(text)
				if val>best.value then
					local targetPart=surfaceGui.Adornee
					if targetPart and targetPart:IsA("BasePart")then
						local dn=surfaceGui:FindFirstChild("DisplayName",true)
						best={value=val,part=targetPart,displayText=(dn and dn.Text)or"Pet",valueText=text}
					end
				end
			end
		end
	end
	if best.part and best.part.Parent then

		local newKey=best.displayText.."|"..best.valueText
		if newKey~=lastBrainrotKey then
			lastBrainrotKey=newKey
			if Hub then
				pcall(function()
					Hub:Notify({Title="New Best Brainrot!",Content=best.displayText.." — "..best.valueText,Duration=4})
				end)
			end
		end
		makeBrainrotBillboard(best.part,best.displayText,best.valueText)
		updateBrainrotBeam(best.part)
	else
		if brainrotCurrentBB then pcall(function()brainrotCurrentBB:Destroy()end);brainrotCurrentBB=nil end
		destroyBrainrotBeam()
	end
end

	if brainrotLoopTask then return end

end

end

local antiRagV2Conns={}
local cleanRagV2Scheduled=false
local function cleanRagV2(char)
	if not char then return end
	for _,d in ipairs(char:GetDescendants())do
		if d:IsA("BallSocketConstraint")or d:IsA("NoCollisionConstraint")or d:IsA("HingeConstraint")
		  or(d:IsA("Attachment")and(d.Name=="A"or d.Name=="B"))
		  or d:IsA("BodyVelocity")or d:IsA("BodyPosition")or d:IsA("BodyGyro")then
			pcall(function()d:Destroy()end)
		end
	end
	for _,d in ipairs(char:GetDescendants())do if d:IsA("Motor6D")then d.Enabled=true end end
	local hum=char:FindFirstChildOfClass("Humanoid")
	if hum then
		local anim=hum:FindFirstChild("Animator")
		if anim then
			for _,track in ipairs(anim:GetPlayingAnimationTracks())do
				local n=track.Animation and track.Animation.Name:lower()or""
				if n:find("rag")or n:find("fall")or n:find("hurt")or n:find("down")then track:Stop(0)end
			end
		end
	end
	task.defer(function()pcall(function()local pm=lp:FindFirstChild("PlayerScripts");if pm then pm=pm:FindFirstChild("PlayerModule")end;if pm then require(pm):GetControls():Enable()end end)end)
end
local function isRagV2Desc(obj)
	return obj:IsA("BallSocketConstraint")or obj:IsA("NoCollisionConstraint")or obj:IsA("HingeConstraint")
	  or(obj:IsA("Attachment")and(obj.Name=="A"or obj.Name=="B"))
	  or obj:IsA("BodyVelocity")or obj:IsA("BodyPosition")or obj:IsA("BodyGyro")
end
local function hookAntiRagV2(char)
	for _,c in ipairs(antiRagV2Conns)do pcall(function()c:Disconnect()end)end;antiRagV2Conns={}
	local hum=char:WaitForChild("Humanoid",10);local hrp=char:WaitForChild("HumanoidRootPart",10)
	if not hum or not hrp then return end
	local lastVel=Vector3.zero
	local c1=hum.StateChanged:Connect(function()
		local st=hum:GetState()
		if st==Enum.HumanoidStateType.Physics or st==Enum.HumanoidStateType.Ragdoll
		  or st==Enum.HumanoidStateType.FallingDown or st==Enum.HumanoidStateType.GettingUp then
			hum:ChangeState(Enum.HumanoidStateType.Running);cleanRagV2(char)
			pcall(function()workspace.CurrentCamera.CameraSubject=hum end)
			task.defer(function()pcall(function()local pm=lp:FindFirstChild("PlayerScripts");if pm then pm=pm:FindFirstChild("PlayerModule")end;if pm then require(pm):GetControls():Enable()end end)end)
		end
	end)
	table.insert(antiRagV2Conns,c1)
	local c2=char.DescendantAdded:Connect(function(desc)
		if isRagV2Desc(desc)and not cleanRagV2Scheduled then
			cleanRagV2Scheduled=true
			task.defer(function()cleanRagV2Scheduled=false;if char and char.Parent then cleanRagV2(char)end end)
		end
	end)
	table.insert(antiRagV2Conns,c2)
	local c3=RS.Heartbeat:Connect(function()
		local st=hum:GetState()
		if st==Enum.HumanoidStateType.Physics or st==Enum.HumanoidStateType.Ragdoll
		  or st==Enum.HumanoidStateType.FallingDown or st==Enum.HumanoidStateType.GettingUp then
			cleanRagV2(char)
			local vel=hrp.AssemblyLinearVelocity
			if(vel-lastVel).Magnitude>40 and vel.Magnitude>25 then hrp.AssemblyLinearVelocity=vel.Unit*math.min(vel.Magnitude,15)end
		end
		lastVel=hrp.AssemblyLinearVelocity
	end)
	table.insert(antiRagV2Conns,c3);cleanRagV2(char)
end

local function getBestBrainrotPart()
	local debris=workspace:FindFirstChild("Debris");if not debris then return nil end
	local best={value=-1,part=nil}
	for _,template in ipairs(debris:GetChildren())do
		if template.Name=="FastOverheadTemplate" then
			local surfaceGui=template:FindFirstChildOfClass("SurfaceGui");if not surfaceGui then continue end
			local genLabel=surfaceGui:FindFirstChild("Generation",true)
			if not genLabel or not genLabel:IsA("TextLabel") then continue end
			local text=genLabel.Text or ""
			if text~="" and (text:find("/s") or text:find("K") or text:find("M") or text:find("B")) then
				local val=parseBrainrotValue(text)
				if val>best.value then
					local targetPart=surfaceGui.Adornee
					if targetPart and targetPart:IsA("BasePart") then
						best={value=val,part=targetPart}
					end
				end
			end
		end
	end
	return best.part
end

local function safeStop(lv, att, hrp)
	pcall(function()
		if lv and lv.Parent then
			lv.VectorVelocity = Vector3.zero
			lv.MaxForce = 0
			task.wait()
			lv:Destroy()
		end
	end)
	pcall(function()
		if att and att.Parent then att:Destroy() end
	end)
	pcall(function()
		if hrp and hrp.Parent then
			hrp.AssemblyLinearVelocity  = Vector3.zero
			hrp.AssemblyAngularVelocity = Vector3.zero
		end
	end)
end

local function wallCheck(hrp, dir, dist)
	if dir.Magnitude == 0 then return false end
	local rp = RaycastParams.new()
	rp.FilterDescendantsInstances = {lp.Character}
	rp.FilterType = Enum.RaycastFilterType.Exclude
	local result = workspace:Raycast(hrp.Position, dir.Unit * dist, rp)
	return result ~= nil
end

local function runCarpetFly()
	if carpetFlying then return end
	if not hasCarpetInInventory() then return end

	carpetFlying = true

	local char = lp.Character
	local hrp  = char and char:FindFirstChild("HumanoidRootPart")
	local hum  = char and char:FindFirstChildOfClass("Humanoid")
	if not hrp or not hum then carpetFlying = false; return end

	local targetPart = getBestBrainrotPart()
	if not targetPart or not targetPart.Parent then carpetFlying = false; return end

	equipCarpet()
	task.wait(0.3)

	char = lp.Character
	hrp  = char and char:FindFirstChild("HumanoidRootPart")
	hum  = char and char:FindFirstChildOfClass("Humanoid")
	if not hrp or not hum then carpetFlying = false; return end

	for _, v in ipairs(hrp:GetChildren()) do
		if v:IsA("LinearVelocity") or v:IsA("BodyVelocity") or v:IsA("BodyPosition")
			or (v:IsA("Attachment") and v.Name == "KnoxCarpetAtt") then
			pcall(function() v:Destroy() end)
		end
	end

	local att = Instance.new("Attachment")
	att.Name  = "KnoxCarpetAtt"
	att.Parent = hrp

	local lv = Instance.new("LinearVelocity")
	lv.Name                  = "KnoxCarpetLV"
	lv.Attachment0           = att
	lv.MaxForce              = 500000
	lv.VelocityConstraintMode = Enum.VelocityConstraintMode.Vector
	lv.VectorVelocity        = Vector3.zero
	lv.RelativeTo            = Enum.ActuatorRelativeTo.World
	lv.Parent                = hrp

	local SPEED      = 110
	local ARRIVE_XZ  = 2.5
	local ARRIVE_Y   = 1.5
	local WALL_DIST  = 3.5
	local SLOW_DIST  = 12
	local dt         = 1/60

	local targetY   = targetPart.Position.Y
	local yTimeout  = tick() + 8

	while carpetFlying and tick() < yTimeout do
		if not hrp or not hrp.Parent then safeStop(lv,att,hrp); carpetFlying=false; return end
		local yDiff = targetY - hrp.Position.Y
		if math.abs(yDiff) <= ARRIVE_Y then break end

		local yDir = Vector3.new(0, math.sign(yDiff), 0)

		if wallCheck(hrp, yDir, WALL_DIST) then
			lv.VectorVelocity = Vector3.zero
			break
		end

		local ySpeed = math.min(SPEED, math.abs(yDiff) * 6)
		ySpeed = math.max(ySpeed, 8)
		lv.VectorVelocity = Vector3.new(0, math.sign(yDiff) * ySpeed, 0)
		task.wait(dt)
	end

	lv.VectorVelocity = Vector3.zero
	if hrp and hrp.Parent then
		hrp.AssemblyLinearVelocity = Vector3.zero
	end
	task.wait(0.05)

	local xzTimeout = tick() + 15

	while carpetFlying and tick() < xzTimeout do
		if not hrp or not hrp.Parent then safeStop(lv,att,hrp); carpetFlying=false; return end

		local fresh = (targetPart.Parent and targetPart.Position) or nil
		if not fresh then break end

		local flatDiff = Vector3.new(fresh.X - hrp.Position.X, 0, fresh.Z - hrp.Position.Z)
		local dist     = flatDiff.Magnitude

		if dist <= ARRIVE_XZ then break end

		local dir = flatDiff.Unit

		if wallCheck(hrp, dir, WALL_DIST) then
			lv.VectorVelocity = Vector3.zero

			for _ = 1, 6 do
				if hrp and hrp.Parent then
					hrp.AssemblyLinearVelocity = hrp.AssemblyLinearVelocity * 0.3
				end
				task.wait(dt)
			end
			if hrp and hrp.Parent then hrp.AssemblyLinearVelocity = Vector3.zero end
			break
		end

		local speed = dist > SLOW_DIST and SPEED or math.max(10, SPEED * (dist / SLOW_DIST))
		lv.VectorVelocity = dir * speed

		local yDriftErr = targetY - hrp.Position.Y
		if math.abs(yDriftErr) > 0.8 then
			lv.VectorVelocity = Vector3.new(
				lv.VectorVelocity.X,
				math.clamp(yDriftErr * 4, -20, 20),
				lv.VectorVelocity.Z
			)
		end

		task.wait(dt)
	end

	safeStop(lv, att, hrp)
	unequipCarpet()
	carpetFlying = false
end

local function doFlashAlign()
	local targetPos, podiumLook = findNearestMarker(); if not targetPos then return end
	local char = lp.Character; if not char then return end
	local hrp = char:FindFirstChild("HumanoidRootPart") or char:FindFirstChild("UpperTorso"); if not hrp then return end

	local attachment = Instance.new("Attachment")
	attachment.Name = "VelocityAttachment"
	attachment.Parent = hrp
	local linearVelocity = Instance.new("LinearVelocity")
	linearVelocity.Name = "TargetLinearVelocity"
	linearVelocity.Attachment0 = attachment
	linearVelocity.MaxForce = 90000
	linearVelocity.VelocityConstraintMode = Enum.VelocityConstraintMode.Vector
	linearVelocity.VectorVelocity = V3(0,0,0)
	linearVelocity.Parent = hrp

	local moveSpeed = 14
	local startTime = os.clock()
	while os.clock() - startTime < 3 do
		local currentPos = hrp.Position
		local direction = V3(targetPos.X-currentPos.X, 0, targetPos.Z-currentPos.Z)
		local distance = direction.Magnitude
		if distance <= 1.5 then break end
		local speed = math.min(moveSpeed, distance * 8)
		linearVelocity.VectorVelocity = direction.Unit * speed
		task.wait()
	end

	linearVelocity.VectorVelocity = V3(0,0,0)
	hrp.AssemblyLinearVelocity = V3(0,0,0)
	linearVelocity:Destroy()
	attachment:Destroy()

	hrp.CFrame = CFrame.new(V3(targetPos.X, hrp.Position.Y, targetPos.Z))
	task.wait(0.1)

	local cam = workspace.CurrentCamera
	if cam then
		local camPos = hrp.Position + V3(0,3,0)
		local flatDir = V3(-podiumLook.X, 0, -podiumLook.Z).Unit
		local upAngle = math.rad(28.4)
		local newLook = (flatDir * math.cos(upAngle) + V3(0,1,0) * math.sin(upAngle)).Unit
		cam.CFrame = CFrame.lookAt(camPos, camPos + newLook)
	end
end

local flashActiveTriggers = {}
local flashPromptConn = nil

local function startFlashTpListener()
	if flashPromptConn then flashPromptConn:Disconnect(); flashPromptConn = nil end
	local PPS = game:GetService("ProximityPromptService")
	flashPromptConn = PPS.PromptButtonHoldBegan:Connect(function(prompt)
		local a = prompt.ActionText and prompt.ActionText:lower() or ""
		local o = prompt.ObjectText  and prompt.ObjectText:lower()  or ""
		if not (a:find("steal") or o:find("steal")) then return end
		if flashActiveTriggers[prompt] then return end
		flashActiveTriggers[prompt] = true
		local startT = os.clock()
		local fired  = false
		local threshold = flashTpPercent / 100
		local conn
		conn = RS.PreRender:Connect(function()
			if not prompt or not prompt.Parent then
				conn:Disconnect(); flashActiveTriggers[prompt] = nil; return
			end

			local prog = math.clamp((os.clock() - startT) / prompt.HoldDuration, 0, 1)
			if not fired and prog >= threshold then
				fired = true
				conn:Disconnect()
				flashActiveTriggers[prompt] = nil
				local char = lp.Character
				if char then
					local tool = char:FindFirstChildOfClass("Tool")
					if tool then pcall(function() tool:Activate() end) end
				end
				if flashTpAutoPoint then task.spawn(ftpUseGiantPotion) end
			end
		end)
		prompt.PromptButtonHoldEnded:Connect(function()
			if not fired then conn:Disconnect(); flashActiveTriggers[prompt] = nil end
		end)
	end)
end

local function stopFlashTpListener()
	if flashPromptConn then flashPromptConn:Disconnect(); flashPromptConn = nil end
	flashActiveTriggers = {}
end

		infJumpConn=UIS.JumpRequest:Connect(function()
			local char=lp.Character;local hrp=char and char:FindFirstChild("HumanoidRootPart")
			if hrp then hrp.AssemblyLinearVelocity=V3(hrp.AssemblyLinearVelocity.X,50,hrp.AssemblyLinearVelocity.Z)end
		end)
	else if infJumpConn then infJumpConn:Disconnect();infJumpConn=nil end end
end})

local antiLagEnabled=false
local antiLagConns={}
local function enableAntiLag()
	if antiLagEnabled then return end
	antiLagEnabled=true
	pcall(function()
		for _,obj in ipairs(workspace:GetDescendants())do
			if obj:IsA("ParticleEmitter")or obj:IsA("Smoke")or obj:IsA("Fire")or obj:IsA("Sparkles")then
				obj.Enabled=false
			elseif obj:IsA("Beam")or obj:IsA("Trail")then
				obj.Enabled=false
			elseif obj:IsA("Sound")then
				obj.Volume=0
			end
		end
	end)
	table.insert(antiLagConns,workspace.DescendantAdded:Connect(function(obj)
		if not antiLagEnabled then return end
		pcall(function()
			if obj:IsA("ParticleEmitter")or obj:IsA("Smoke")or obj:IsA("Fire")or obj:IsA("Sparkles")then
				obj.Enabled=false
			elseif obj:IsA("Beam")or obj:IsA("Trail")then
				obj.Enabled=false
			end
		end)
	end))
	pcall(function()
		settings().Rendering.QualityLevel=Enum.QualityLevel.Level01
	end)
	pcall(function()
		workspace.StreamingEnabled=true
	end)
end
local function disableAntiLag()
	if not antiLagEnabled then return end
	antiLagEnabled=false
	for _,c in ipairs(antiLagConns)do pcall(function()c:Disconnect()end)end
	antiLagConns={}
end
VisTab:CreateToggle({Name="Anti Lag",Flag="AntiLag",CurrentValue=false,Callback=function(state)
	if state then enableAntiLag()else disableAntiLag()end
end})

local function resetMiniButtonPositions()
	local MINI_POS_PREFIX="__minipos__"
	local entries={
		{isAction=false,flag="AutoGrab"},
		{isAction=false,flag="Booster"},
		{isAction=false,flag="InfJump"},
		{isAction=false,flag="AntiRagV2"},
	}
	for _,entry in ipairs(entries)do
		if entry.isAction then
			Ascended.Flags[MINI_POS_PREFIX.."act__"..entry.flag]=nil
		else
			Ascended.Flags[MINI_POS_PREFIX..entry.flag]=nil
		end
	end
	Ascended.Flags["SaveConfigToggle"]=false
	Hub.Config.Enabled=true
	Hub:Save()
	Hub.Config.Enabled=false
	Hub:Notify({
		Title="Positions Reset",
		Content="Turn Off Save Config, Click ResetMiniBtn, Rejoin.",
		Duration=5
	})
end

ExtraTab:CreateButton("ResetMiniBtn",function()
	resetMiniButtonPositions()
end)
